"use strict";
function errorHandler(mesg) {
    const code = _getCode(mesg);
    this.throw(code);
}
exports.errorHandler = errorHandler;
function _getCode(mesg) {
    let code;
    switch (mesg) {
        case 'bad request':
            code = 400;
            break;
        case 'unauthorized':
            code = 401;
            break;
        case 'forbidden':
            code = 403;
            break;
        default:
            code = 500;
            break;
    }
    return code;
}
